function void soulCode(S:string,N){

    if(N!=1){
        C[1,array] = S:explode(".")
        #printTable(C)
    }else{
        local S = S:explode(";")
        for(I=1,S:count()){
            C[I,array]=S[I,string]:trim():explode(".")

                Sorted = 1
            }
        }
    }
    if(Sorted == 1){
        #printTable(C)
        if(C[1,array][1,string]=="soulStart"){
